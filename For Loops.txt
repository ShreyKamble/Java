class JavaClass {
    public static void main(String[] args) {
    	for(int i = 0; i<=10; i++) {
    		System.out.println(i);
    	}
    }
}


// we can also use "counter" keyword.

class JavaClass {
    public static void main(String[] args) {
    	for(int counter = 0; counter <= 10; counter++) {
    		System.out.println("Hello World");
    	}
    }
}
