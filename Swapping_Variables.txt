public class JavaClass {
	public static void main (String[] args){
		int a = 20;
		int b = 30;
		int temp;
		//Before Swapping
		System.out.println(a);
		System.out.println(b);
		
		temp = a;
		a = b;
		b = temp;
		//After Swapping
		System.out.println(a);
		System.out.println(b);
	}
}
